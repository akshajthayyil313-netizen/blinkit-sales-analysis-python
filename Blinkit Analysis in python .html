import matplotlib.pyplot as plt

# 1. Calculate the sales data
sales_by_type = df.groupby('Item Type')['Sales'].sum().sort_values(ascending=False)

# 2. Set up the figure size (This must be on its own line)
plt.figure(figsize=(10, 6))

# 3. Create the bar chart (This must be on its own line)
bars = plt.bar(sales_by_type.index, sales_by_type.values)

# 4. Apply labels and formatting
plt.xticks(rotation=-90)
plt.xlabel('Item Type')
plt.ylabel('Total Sales')
plt.title('Total Sales by Item Type')

# 5. Add the data numbers on top of each bar
for bar in bars:
    plt.text(bar.get_x() + bar.get_width() / 2, 
             bar.get_height(), 
             f'{bar.get_height():,.0f}', 
             ha='center', 
             va='bottom')

# 6. Display the final plot
plt.tight_layout()
plt.show()

import seaborn as sns

sales_by_location = df.groupby('Outlet Location Type')['Sales'].sum().reset_index()
sales_by_location = sales_by_location.sort_values('Sales', ascending=False)

# 2. Setup the figure
plt.figure(figsize=(8, 3))

# 3. Create the plot
# IMPORTANT: This must be on a new line. Do not put it after a '#' symbol.

plt.title('Total Sales by Outlet Location Type')
plt.xlabel('Total Sales')
plt.ylabel('Outlet Location Type')
